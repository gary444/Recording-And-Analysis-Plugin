82struct Checkpoint {
83 float time = -1.0f;
84 long long already_read_elements = -1;
85 std::map<int, std::shared_ptr<RecordableData>> last_seen_data;
86
87 Checkpoint(float t, long long read_elements, std::map<int, std::shared_ptr<RecordableData>> const& data) : time(t), already_read_elements(read_elements){
88 for(auto const & element : data){
89 last_seen_data[element.first] = element.second;
90 }
91 }
92
93 ~Checkpoint(){
94 for(auto& element : last_seen_data){
95 if(element.second != nullptr) {
96 element.second = nullptr;
97 } else {
98 Debug::Log("Error! checkpoint contained invalid data!", Color::Red);
99 }
100 }
101
102 last_seen_data.clear();
103 }
104};
